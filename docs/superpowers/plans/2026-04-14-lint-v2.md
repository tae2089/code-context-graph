// 5. Find contradictions: annotation has @param but node was updated after annotation.
{
	// Query annotations with their tags for all symbol node IDs
	var anns []model.Annotation
	if len(ids) > 0 {
		g.DB.Where("node_id IN ?", ids).Preload("Tags").Find(&anns)
	}
	for _, a := range anns {
		hasParam := false
		for _, tag := range a.Tags {
			if tag.Kind == model.TagParam {
				hasParam = true
				break
			}
		}
		if !hasParam {
			continue
		}
		n := nodeByID[a.NodeID]
		if n == nil {
			continue
		}
		// Reload node to get accurate UpdatedAt
		var freshNode model.Node
		if err := g.DB.Select("updated_at").First(&freshNode, n.ID).Error; err != nil {
			continue
		}
		if freshNode.UpdatedAt.After(a.UpdatedAt) {
			report.Contradictions = append(report.Contradictions, Contradiction{
				QualifiedName: n.QualifiedName,
				Detail:        "@param exists but node updated since annotation",
			})
		}
	}
	sort.Slice(report.Contradictions, func(i, j int) bool {
		return report.Contradictions[i].QualifiedName < report.Contradictions[j].QualifiedName
	})
}

// internal/docs/history.go
package docs

import (
	"encoding/json"
	"os"
	"sort"
	"time"
)

// History tracks lint results across runs for the Twice Rule.
type History struct {
	Timestamp time.Time      `json:"timestamp"`
	Entries   map[string]int `json:"entries"` // "category:qualified_name" → consecutive count
}

// LoadHistory reads the history file. Returns empty history if file doesn't exist.
func LoadHistory(path string) (*History, error) {
	h := &History{Entries: map[string]int{}}

	data, err := os.ReadFile(path)
	if err != nil {
		if os.IsNotExist(err) {
			return h, nil
		}
		return nil, err
	}

	if err := json.Unmarshal(data, h); err != nil {
		return nil, err
	}
	if h.Entries == nil {
		h.Entries = map[string]int{}
	}
	return h, nil
}

// Update compares current lint keys against stored history.
// Returns keys that reached count >= 2 (Twice Rule triggered).
// Mutates h.Entries in place.
func (h *History) Update(currentKeys []string) []string {
	currentSet := map[string]bool{}
	for _, k := range currentKeys {
		currentSet[k] = true
	}

	var triggered []string

	// Increment existing or mark resolved
	for key := range h.Entries {
		if !currentSet[key] {
			delete(h.Entries, key) // resolved
		}
	}

	for _, key := range currentKeys {
		h.Entries[key]++
		if h.Entries[key] >= 2 {
			triggered = append(triggered, key)
		}
	}

	h.Timestamp = time.Now()
	sort.Strings(triggered)
	return triggered
}

// Save writes the history to the given path, creating parent dirs if needed.
func (h *History) Save(path string) error {
	if err := os.MkdirAll(filepath.Dir(path), 0o755); err != nil {
		return err
	}
	data, err := json.MarshalIndent(h, "", "  ")
	if err != nil {
		return err
	}
	return os.WriteFile(path, data, 0o644)
}

// Rule represents a lint rule entry in .ccg.yaml.
type Rule struct {
	Pattern  string `yaml:"pattern"`
	Category string `yaml:"category"`
	Action   string `yaml:"action"`
	Auto     bool   `yaml:"auto"`
	Created  string `yaml:"created"`
}

// WriteYamlRules appends Twice-Rule-triggered entries to .ccg.yaml rules section.
// Idempotent: skips rules whose pattern+category already exist.
func WriteYamlRules(cfgPath string, triggered []string) error {
	// Read existing file
	data, err := os.ReadFile(cfgPath)
	if err != nil && !os.IsNotExist(err) {
		return err
	}

	content := string(data)

	// Parse triggered keys ("category:qualified_name")
	var newRules []Rule
	for _, key := range triggered {
		parts := strings.SplitN(key, ":", 2)
		if len(parts) != 2 {
			continue
		}
		category, pattern := parts[0], parts[1]

		// Idempotency: skip if pattern already in file
		if strings.Contains(content, pattern) {
			continue
		}

		newRules = append(newRules, Rule{
			Pattern:  pattern,
			Category: category,
			Action:   "warn",
			Auto:     true,
			Created:  time.Now().Format("2006-01-02"),
		})
	}

	if len(newRules) == 0 {
		return nil
	}

	// Build YAML snippet to append
	var sb strings.Builder
	if !strings.Contains(content, "rules:") {
		sb.WriteString("\nrules:\n")
	}
	for _, r := range newRules {
		sb.WriteString(fmt.Sprintf("  - pattern: %q\n", r.Pattern))
		sb.WriteString(fmt.Sprintf("    category: %s\n", r.Category))
		sb.WriteString(fmt.Sprintf("    action: %s\n", r.Action))
		sb.WriteString(fmt.Sprintf("    auto: true\n"))
		sb.WriteString(fmt.Sprintf("    created: %q\n", r.Created))
	}

	content += sb.String()
	return os.WriteFile(cfgPath, []byte(content), 0o644)
}
