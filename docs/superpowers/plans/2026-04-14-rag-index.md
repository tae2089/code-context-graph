# ccg Vectorless RAG Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** docs/ 디렉토리와 community 구조에서 LLM 없이 트리 인덱스를 빌드하고, MCP 툴 3개(`build_rag_index`, `get_rag_tree`, `get_doc_content`)와 CLI `ccg rag-index`로 AI 에이전트가 문서를 탐색할 수 있게 한다.

**Architecture:** `ragindex.Builder`가 DB의 Community + CommunityMembership + Annotation/DocTag를 읽어 `.ccg/doc-index.json`을 생성한다. CLI와 MCP 핸들러 모두 동일한 Builder를 호출하며, `get_rag_tree`와 `get_doc_content`는 기존 MCP 캐시 패턴을 그대로 적용한다.

**Tech Stack:** Go 1.24, GORM (sqlite), cobra, mcp-go (`github.com/mark3labs/mcp-go`), `encoding/json`, `os`

package ragindex

import (
    "encoding/json"
    "fmt"
    "os"
    "path/filepath"
    "time"

    "gorm.io/gorm"

    "github.com/imtaebin/code-context-graph/internal/model"
)

// TreeNode는 doc-index.json의 단일 노드이다.
type TreeNode struct {
    ID       string      `json:"id"`
    Label    string      `json:"label"`
    Summary  string      `json:"summary"`
    DocPath  string      `json:"doc_path,omitempty"` // file 노드만 설정
    Children []*TreeNode `json:"children"`
}

// Index는 .ccg/doc-index.json 전체 포맷이다.
type Index struct {
    Version int       `json:"version"`
    BuiltAt time.Time `json:"built_at"`
    Root    *TreeNode `json:"root"`
}

// Builder는 DB에서 인덱스를 빌드하는 구조체이다.
type Builder struct {
    DB      *gorm.DB
    OutDir  string // docs 디렉토리 (기본: "docs")
    IndexDir string // .ccg 디렉토리 (기본: ".ccg")
}

// Build는 DB에서 트리를 생성하고 .ccg/doc-index.json에 저장한다.
// 반환값: (community 수, file 노드 수, error)
func (b *Builder) Build() (int, int, error) {
    root, communities, files, err := b.buildTree()
    if err != nil {
        return 0, 0, err
    }

    idx := &Index{
        Version: 1,
        BuiltAt: time.Now().UTC(),
        Root:    root,
    }

    if err := b.writeIndex(idx); err != nil {
        return 0, 0, err
    }

    return communities, files, nil
}

func (b *Builder) buildTree() (*TreeNode, int, int, error) {
    // 1. 모든 community 로드
    var communities []model.Community
    if err := b.DB.Preload("Members").Find(&communities).Error; err != nil {
        return nil, 0, 0, fmt.Errorf("load communities: %w", err)
    }

    root := &TreeNode{
        ID:       "root",
        Label:    "root",
        Summary:  "",
        Children: make([]*TreeNode, 0),
    }

    totalFiles := 0

    for _, c := range communities {
        communityNode := &TreeNode{
            ID:       fmt.Sprintf("community:%d", c.ID),
            Label:    c.Label,
            Summary:  c.Description,
            Children: make([]*TreeNode, 0),
        }

        // 2. 멤버 nodeID 수집
        nodeIDs := make([]uint, 0, len(c.Members))
        for _, m := range c.Members {
            nodeIDs = append(nodeIDs, m.NodeID)
        }

        if len(nodeIDs) == 0 {
            root.Children = append(root.Children, communityNode)
            continue
        }

        // 3. file_path별로 노드 그룹화
        var nodes []model.Node
        if err := b.DB.Where("id IN ?", nodeIDs).Find(&nodes).Error; err != nil {
            return nil, 0, 0, fmt.Errorf("load nodes for community %d: %w", c.ID, err)
        }

        filePaths := make(map[string]bool)
        for _, n := range nodes {
            if n.FilePath != "" {
                filePaths[n.FilePath] = true
            }
        }

        for fp := range filePaths {
            summary := b.fileSummary(fp)
            docPath := filepath.Join(b.outDir(), fp+".md")
            fileNode := &TreeNode{
                ID:       "file:" + fp,
                Label:    filepath.Base(fp),
                Summary:  summary,
                DocPath:  docPath,
                Children: []*TreeNode{},
            }
            communityNode.Children = append(communityNode.Children, fileNode)
            totalFiles++
        }

        root.Children = append(root.Children, communityNode)
    }

    return root, len(communities), totalFiles, nil
}

// fileSummary는 @index 태그 → @intent 태그 → "" 순으로 summary를 추출한다.
func (b *Builder) fileSummary(filePath string) string {
    // @index 태그 조회: 해당 파일 노드들의 어노테이션에서 TagIndex 찾기
    var tag model.DocTag
    err := b.DB.Joins("JOIN annotations ON annotations.id = doc_tags.annotation_id").
        Joins("JOIN nodes ON nodes.id = annotations.node_id").
        Where("nodes.file_path = ? AND doc_tags.kind = ?", filePath, model.TagIndex).
        Order("doc_tags.ordinal ASC").
        First(&tag).Error
    if err == nil {
        return tag.Value
    }

    // fallback: @intent 태그
    err = b.DB.Joins("JOIN annotations ON annotations.id = doc_tags.annotation_id").
        Joins("JOIN nodes ON nodes.id = annotations.node_id").
        Where("nodes.file_path = ? AND doc_tags.kind = ?", filePath, model.TagIntent).
        Order("doc_tags.ordinal ASC").
        First(&tag).Error
    if err == nil {
        return tag.Value
    }

    return ""
}

func (b *Builder) outDir() string {
    if b.OutDir != "" {
        return b.OutDir
    }
    return "docs"
}

func (b *Builder) indexDir() string {
    if b.IndexDir != "" {
        return b.IndexDir
    }
    return ".ccg"
}

func (b *Builder) writeIndex(idx *Index) error {
    dir := b.indexDir()
    if err := os.MkdirAll(dir, 0755); err != nil {
        return fmt.Errorf("create index dir: %w", err)
    }
    path := filepath.Join(dir, "doc-index.json")
    f, err := os.Create(path)
    if err != nil {
        return fmt.Errorf("create doc-index.json: %w", err)
    }
    defer f.Close()
    enc := json.NewEncoder(f)
    enc.SetIndent("", "  ")
    return enc.Encode(idx)
}

package cli

import (
    "fmt"

    "github.com/spf13/cobra"

    "github.com/imtaebin/code-context-graph/internal/ragindex"
)

func newRagIndexCmd(deps *Deps) *cobra.Command {
    var outDir string
    var indexDir string

    cmd := &cobra.Command{
        Use:   "rag-index",
        Short: "Build Vectorless RAG index from docs and community structure",
        Args:  cobra.NoArgs,
        RunE: func(cmd *cobra.Command, args []string) error {
            if deps.DB == nil {
                return fmt.Errorf("database not initialized")
            }

            b := &ragindex.Builder{
                DB:       deps.DB,
                OutDir:   resolveOutDir(outDir),
                IndexDir: indexDir,
            }

            communities, files, err := b.Build()
            if err != nil {
                return fmt.Errorf("build rag index: %w", err)
            }

            fmt.Fprintf(stdout(cmd), "Built doc-index: %d communities, %d files\n", communities, files)
            return nil
        },
    }

    cmd.Flags().StringVar(&outDir, "out", "docs", "Documentation directory")
    cmd.Flags().StringVar(&indexDir, "index-dir", ".ccg", "Directory for doc-index.json output")

    return cmd
}

func (h *handlers) getRagTree(ctx context.Context, request mcp.CallToolRequest) (*mcp.CallToolResult, error) {
    communityID, _ := request.GetString("community_id")

    key := "get_rag_tree:" + mustJSON(map[string]any{"community_id": communityID})
    if h.cache != nil {
        if cached, ok := h.cache.Get(key); ok {
            return mcp.NewToolResultText(cached), nil
        }
    }

    idx, err := ragindex.LoadIndex(".ccg/doc-index.json")
    if err != nil {
        return mcp.NewToolResultError(fmt.Sprintf("load doc-index: %v", err)), nil
    }

    var node *ragindex.TreeNode
    if communityID == "" {
        node = idx.Root
    } else {
        node = ragindex.FindNode(idx.Root, communityID)
        if node == nil {
            return mcp.NewToolResultError(fmt.Sprintf("community_id %q not found", communityID)), nil
        }
    }

    b, err := json.Marshal(node)
    if err != nil {
        return mcp.NewToolResultError(fmt.Sprintf("marshal tree: %v", err)), nil
    }
    result := string(b)
    if h.cache != nil {
        h.cache.Set(key, result)
    }
    return mcp.NewToolResultText(result), nil
}

func (h *handlers) getDocContent(ctx context.Context, request mcp.CallToolRequest) (*mcp.CallToolResult, error) {
    filePath, err := request.RequireString("file_path")
    if err != nil {
        return mcp.NewToolResultError(fmt.Sprintf("missing parameter: %v", err)), nil
    }

    key := "get_doc_content:" + mustJSON(map[string]any{"file_path": filePath})
    if h.cache != nil {
        if cached, ok := h.cache.Get(key); ok {
            return mcp.NewToolResultText(cached), nil
        }
    }

    // 경로 traversal 방지: filepath.Clean 후 ".." 포함 여부 검사
    clean := filepath.Clean(filePath)
    if strings.HasPrefix(clean, "..") {
        return mcp.NewToolResultError("invalid file_path: path traversal not allowed"), nil
    }

    content, err := os.ReadFile(clean)
    if err != nil {
        return mcp.NewToolResultError(fmt.Sprintf("read file %q: %v", filePath, err)), nil
    }
    result := string(content)
    if h.cache != nil {
        h.cache.Set(key, result)
    }
    return mcp.NewToolResultText(result), nil
}
